before

Adiciona elemento antes da tag alvo 
<div id="exemplo">
    <div class="meio">meio</div>
</div>

Adesão por elemento criado pelo javascript

var p = document.createElement('p');
p.innerText = 'Função before por script externo';
jsd('#exemplo').before(p);

Adesão por elemento criado pelo create

var p = jsd.create({tagName: 'p'});
p.innerText = 'Função create seguida de before';
jsd('#exemplo').before(p);

Adiciona e ao mesmo tempo cria o elemento

jsd('#exemplo').before('<p>Adiciona e cria usando texto</p>');

Move os elementos

var mover = document.getElementsByClassName('meio');
jsd('#exemplo').before(mover);

Move os elementos

var mover = jsd('.meio');
jsd('#exemplo').before(mover);
