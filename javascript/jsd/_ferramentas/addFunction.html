addFunction

Adiciona nova função vinculada a jsd
Adicionar alguma função 
function funcaoA() {
    console.log('Função A', this('#exemplo').html());
}
jsd.addFunction(funcaoA);

// Executa função adicionada
jsd.funcaoA();
Adicionar multiplas funções 
function boooo() {
    console.log('BOOOOOOO!!!!', this);
}
jsd.addFunction({
    funcaoB: function () {
    console.log('Função B');
    },
    funcaoC: function () {
        console.log('Função C');
    },
    funcaoD: boooo
});

// Executa funções adicionadas
jsd.funcaoB();
jsd.funcaoC();
jsd.funcaoD();
