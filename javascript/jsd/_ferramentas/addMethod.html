addMethod

Adiciona novo método vinculada os elementos
Adicionar algum método 
function metodoA(param) {
    console.log(
        'Método A',
        this[0].innerHTML = param,
        arguments
    );
}
jsd.addMethod(metodoA);

// Executa o método
jsd('#exemplo').metodoA('Parametro');
Adicionar multiplos métodos 
function boooo() {
    console.log('BOOOOOOO!!!!', this);
}
jsd.addMethod({
    metodoB: function () {
        console.log('Método B');
    },
    metodoC: function () {
        console.log('Método C');
    },
    metodoD: boooo
});

// Executa métodos adicionados
jsd('#exemplo').metodoB();
jsd('#exemplo').metodoC();
jsd('#exemplo').metodoD();
