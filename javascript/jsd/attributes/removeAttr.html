removeAttr

Remove atributos do elemento

recursion

Esse método é recursivo outros elementos

jsd('#exemplo').removeAttr(); // Executa apenas em <div id="exemplo">
jsd('.exemplo').removeAttr(); // Executa em todos <div class="exemplo">
jsd('#exemplo').removeAttr('style'); // Executa apenas em <div id="exemplo">
jsd('.exemplo').removeAttr('style'); // Executa em todos <div class="exemplo">
<div id="exemplo" style="color:black; padding:10px" data-exemplo="data valor">Exemplo</div>
