same

Faz a comparação de algum elemento possa ser o mesmo que o outro. 
<div id="id-a" class="class-a"></div>
<div id="id-b" class="class-a"></div>
<div id="id-c" class="class-b"></div>

Comparando por texto

console.log(
    jsd('#id-a').same('.class-a'), // true
    jsd('#id-b').same('.class-a'), // true
    jsd('#id-a').same('.class-b'), // false
    jsd('#id-b').same('.class-b'), // false
    jsd('.class-a').same('#id-a')  // true
);

Usando variáveis


var divID = document.getElementById('id-a');
var divClass = document.getElementsByClassName('class-a');

var divID_2 = jsd('#id-a');
var divClass_2 = jsd('.class-a');

console.log(
    jsd(divID).same(divClass), // true
    divID_2.same(divClass),    // true
    divID_2.same(divClass_2),  // true
    divClass_2.same(divID)     // true
);
