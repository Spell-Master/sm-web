Remove ouvintes de eventos adicionados. 
<button id="exemplo">Exemplo</button>

<script>
    function funcaoDefinida() {
        console.log('Evento mousedown');
    }

    var exemplo = jsd('#exemplo');

    exemplo.on('click', function (e) {
        console.log('Evento click');
    });

    exemplo.on('mousedown', funcaoDefinida);

    exemplo.on('dblclick', function () {
        console.log('Evento dblclick');
    });
</script>
Remove ouvinte click 
exemplo.off('click');
Remove ouvinte que espera a função expecífica 
exemplo.off('funcaoDefinida');
Remove todos eventos 
exemplo.off();
