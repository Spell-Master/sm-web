Esse método é recursivo outros elementos

jsd('#exemplo').css(); // Executa apenas em <div id="exemplo">
jsd('.exemplo').css(); // Executa apenas no primeiro <div class="exemplo">

jsd('#exemplo').css('color'); // Executa apenas em <div id="exemplo">
jsd('.exemplo').css('color'); // Executa apenas no primeiro <div class="exemplo">

jsd('#exemplo').css({color: red}); // Executa apenas em <div id="exemplo">
jsd('.exemplo').css({color: red}); // Executa em todos <div class="exemplo">
