setData

Manipula pseudo atributo data

recursion

Esse método é recursivo outros elementos

jsd('#exemplo').setData({a: 'Valor de A'}); // Executa apenas em <div id="exemplo">
jsd('.exemplo').setData({a: 'Valor de A'}); // Executa em todos <div class="exemplo">
<div id="exemplo">Exemplo</div>
<script>
    jsd('#exemplo').setData({
        exemplo: 'valor de exemplo',
        'outro-data': 'valor de outro',
        'vazio': '' // nenhum valor
    });
</script>
