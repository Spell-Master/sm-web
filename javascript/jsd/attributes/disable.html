disable

Habilita ou desabilita elemento

recursion

Esse método é recursivo outros elementos

jsd('#exemplo').disable(); // Executa apenas em <div id="exemplo">
jsd('.exemplo').disable(); // Executa apenas no primeiro <div class="exemplo">
jsd('#exemplo').disable(true); // Executa apenas em <div id="exemplo">
jsd('.exemplo').disable(true); // Executa em todos <div class="exemplo">
jsd('#exemplo').disable('toggle'); // Executa apenas em <div id="exemplo">
jsd('.exemplo').disable('toggle'); // Executa em todos <div class="exemplo">
<input type="text" value="exemplo" class="input-default" />
