clear

Elimina elementos dentro do alvo

recursion

Esse método é recursivo outros elementos

jsd('#exemplo').clear(); // Executa apenas em <div id="exemplo">
jsd('.exemplo').clear(); // Executa em todos <div class="exemplo">
