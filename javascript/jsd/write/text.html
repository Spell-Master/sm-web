recursion

Esse método é recursivo outros elementos

jsd('#exemplo').text('Olá mundo'); // Executa apenas em <div id="exemplo">
jsd('.exemplo').text('Olá mundo'); // Executa em todos <div class="exemplo">
jsd('#exemplo').text(); // Executa apenas em <div id="exemplo">
jsd('.exemplo').text(); // Executa apenas no primeiro <div class="exemplo">
<div id="exemplo"><p>exemplo</p></div>
Retorna o texto sem HTML 
jsd('#exemplo').html();
Escreve como texto 
var escrita = '<div>Olá mundo</div>';
jsd('#exemplo').html(escrita);
