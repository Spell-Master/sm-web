each

Percorre grupo de dados
Array 
var array = ['a', 'b', 'c'];
jsd.each(array, function (valor, indice) {
    console.log(
        'indice: ' + indice + '\n'
        + 'valor: ' + valor + '\n'
        + 'retorno: ' + this + '\n'
    );
});
Objeto 
var objeto = {a: 'valor de a', b: 'valor de b', c: 'valor de c'};
jsd.each(objeto, function (valor, indice) {
    console.log(
        'indice: ' + indice + '\n'
        + 'valor: ' + valor + '\n'
        + 'retorno: ' + this
    );
});
