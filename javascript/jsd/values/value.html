    <textarea name="textarea-name">valor do textarea</textarea>

    <select name="select-name">

Botão

Obtem o valor do elemento 
jsd('input[type="text"]').value(); //valor do input
jsd('select').value(); //valor do seletor
Obtem o valor de tudo no formulário

Se a tag alvo for um form altomaticamente retorna todos os values do formulário

jsd('#formulario').value();
Altera o valor 
jsd('textarea').value('Olá mundo');
Não é possível alterar todos os valores diretamente do formulário dessa forma: 
jsd('#formulario').value('Alguma coisa');
Use formElements: 
var formulario = jsd('#formulario');
console.log( formulario.formElements() ); // Vendo todos elementos válidos do formulário
formulario.formElements().each(function (elemento) {
    elemento.value('Outro Valor'); // Redefinindo todos values
});
