prepend

Adiciona elemento interno como primeira tag no alvo 
<div id="exemplo">
    <div class="antes">antes a</div>
    <div class="antes">antes b</div>
    <div class="meio">meio</div>
    <div class="depois">depois a</div>
    <div class="depois">depois b</div>
</div>

Adesão por elemento criado pelo javascript

var p = document.createElement('p');
p.innerText = 'Função prepend por script externo';
jsd('#exemplo').prepend(p);

Adesão por elemento criado pelo create

var p = jsd.create({tagName: 'p'});
p.innerText = 'Função create seguida de prepend';
jsd('#exemplo').prepend(p);

Adiciona e ao mesmo tempo cria o elemento

jsd('#exemplo').prepend('<p>Adiciona e cria usando texto</p>');

Move os elementos

var mover = document.getElementsByClassName('depois');
jsd('#exemplo').prepend(mover);

Move os elementos

var mover = jsd('.depois');
jsd('#exemplo').prepend(mover);
