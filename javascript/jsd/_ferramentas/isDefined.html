isDefined

Verificar se está definido
NOTA*

Se o valor não seja indefinido retorna verdadeiro para função

<div id="exemplo-a"></div>
<script>
    var a = true;
    var b = false;
    var c = null;
    var d = undefined;
    var e = document.getElementById('exemplo-a');
    var f = document.getElementById('exemplo-b');
    // Não existe: <div id="exemplo-b"></div>,
    // mas o retorno de getElementById é null portanto não é valor indefinido
            
    console.log(
        jsd.isDefined(a), // true
        jsd.isDefined(b), // true
        jsd.isDefined(c), // true
        jsd.isDefined(d), // false
        jsd.isDefined(e),  // true
        jsd.isDefined(f),  // true
        
        jsd.typeof(e)
    );
</script>
