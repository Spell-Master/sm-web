after

Adiciona elemento depois da tag alvo 
<div id="exemplo">
    <div class="meio">meio</div>
</div>

Adesão por elemento criado pelo javascript

var p = document.createElement('p');
p.innerText = 'Função after por script externo';
jsd('#exemplo').after(p);

Adesão por elemento criado pelo create

var p = jsd.create({tagName: 'p'});
p.innerText = 'Função create seguida de after';
jsd('#exemplo').after(p);

Adiciona e ao mesmo tempo cria o elemento

jsd('#exemplo').after('<p>Adiciona e cria usando texto</p>');

Move os elementos

var mover = document.getElementsByClassName('meio');
jsd('#exemplo').after(mover);

Move os elementos

var mover = jsd('.meio');
jsd('#exemplo').after(mover);
